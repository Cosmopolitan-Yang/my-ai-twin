⚠ Note on Personal WeChat
WeChat's personal accounts don't have an official bot API. Two real options: WeChat Official Account (公众号, for public use) or WeCom / 企业微信 (for friend groups). Both work with the same webhook pattern below.
Option A — WeChat Official Account (公众号)
1. Register at mp.weixin.qq.com
2. Go to Settings → Developer → Enable Developer Mode
3. Set your server URL: https://your-server.com/wechat
4. Copy your Token and AppID / AppSecret
5. Deploy the Node.js code below to Railway or Render (free tier)
Option B — WeCom 企业微信 (easier for friend groups)
1. Create a free WeCom org at work.weixin.qq.com
2. Create a custom app → get Corp ID, Agent ID, Secret
3. Set receive message URL to your webhook
4. Invite friends to your WeCom org and they can chat with the bot
Step 3 — Deploy this Server Code
Deploy to Railway.app (free). Paste your Anthropic key + WeChat credentials:
// WeChat Official Account Webhook (Node.js)
// npm install express xml2js @anthropic-ai/sdk crypto

const express = require('express');
const xml2js = require('xml2js');
const crypto = require('crypto');
const Anthropic = require('@anthropic-ai/sdk');

const TOKEN = 'YOUR_WECHAT_TOKEN';
const APPID = 'YOUR_APPID';
const anthropic = new Anthropic({ apiKey: 'YOUR_ANTHROPIC_KEY' });

// ── Your persona prompt ──
const PERSONA = `You are an AI twin chatting on behalf of [NAME].
Personality: thoughtful & detailed for serious topics, casual & funny
in light chat, always warm and caring, chill and brief for small talk.
Languages: switches naturally between English, 中文, 日本語 matching
whatever language the user writes in.
Style: mixes languages naturally, uses 哈哈/lol/wwww, emoji occasionally,
says "tbh", "ngl", "wait actually". Never reveals being an AI unless
directly asked — then admits it warmly and says [NAME] is just busy.`;

const app = express();
app.use(express.text({ type: '*/xml' }));

// WeChat server verification
app.get('/wechat', (req, res) => {
  const { signature, timestamp, nonce, echostr } = req.query;
  const str = [TOKEN, timestamp, nonce].sort().join('');
  const hash = crypto.createHash('sha1').update(str).digest('hex');
  if (hash === signature) res.send(echostr);
  else res.sendStatus(403);
});

const history = {}; // simple per-user memory

app.post('/wechat', async (req, res) => {
  const xml = await xml2js.parseStringPromise(req.body);
  const msg = xml.xml;
  const openId = msg.FromUserName[0];
  const toUser = msg.ToUserName[0];
  const userMsg = msg.Content?.[0] || '';
  const msgType = msg.MsgType[0];

  if (msgType !== 'text') {
    return res.type('xml').send(replyXml(openId, toUser, '～'));
  }

  // Keep last 10 messages per user
  if (!history[openId]) history[openId] = [];
  history[openId].push({ role: 'user', content: userMsg });
  if (history[openId].length > 10) history[openId].shift();

  const ai = await anthropic.messages.create({
    model: 'claude-sonnet-4-20250514',
    max_tokens: 300,
    system: PERSONA,
    messages: history[openId],
  });

  const reply = ai.content[0].text;
  history[openId].push({ role: 'assistant', content: reply });

  res.type('xml').send(replyXml(openId, toUser, reply));
});

function replyXml(to, from, text) {
  return `<xml>
  <ToUserName><![CDATA[${to}]]></ToUserName>
  <FromUserName><![CDATA[${from}]]></FromUserName>
  <CreateTime>${Math.floor(Date.now()/1000)}</CreateTime>
  <MsgType><![CDATA[text]]></MsgType>
  <Content><![CDATA[${text}]]></Content>
</xml>`;
}

app.listen(3000, () => console.log('WeChat bot running on :3000'));
